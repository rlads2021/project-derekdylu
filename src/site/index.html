    cwn = CwnBase()

def all_sense_tree (word, verbal = False):
    cnt = 0
    for i in range(len(word)):
        snese_tree = word[i].senses
        cnt += len(word[i].senses)
        if(verbal == True): print(snese_tree)

    if(verbal == True): print("total senses = ", cnt)
    return cnt
